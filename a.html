Lỗi mới phát hiện trong V5

#01 LED IR vẽ ngược cực — đèn không sáng Critical
SVG V5: tam giác LED đỉnh hướng lên, vạch Cathode đặt phía trên, dây trên qua trở 100Ω lên 5V. Cathode lên 5V, Anode xuống Collector → LED phân cực ngược → không bao giờ sáng. Tính năng nhìn đêm bị vô hiệu hóa.
V6: Đảo chiều LED. Đáy tam giác (Anode) ở trên nối 5V qua trở 100Ω. Đỉnh + vạch Cathode ở dưới nối Collector S8050.
#02 Diode Wired-OR vẫn sai chiều Critical
V5 tuyên bố đã sửa nhưng SVG thực tế Anode (x=825) phía module, Cathode (x=837) phía GPIO33. Logic Active-LOW: module kéo OUT xuống 0V → Anode=0V, Cathode=3.3V → diode reverse-biased không dẫn → GPIO 33 luôn HIGH → tia hồng ngoại bị ngắt cũng không trigger.
V6: Anode (đáy tam giác) về GPIO 33. Cathode (vạch) về OUT module. Khi module LOW: dòng chạy từ 3.3V → pull-up → Anode → Cathode → GND module.
#03 Trở 100Ω chưa ghi công suất Warning
P thực = 0.038² × 100 = 144mW mỗi trở. Trở 1/8W (125mW) bị quá tải 15% → đen, khói, cháy lan PCB sau vài giờ chạy liên tục.
V6: Bắt buộc trở 100Ω 1/4W (250mW) trở lên. Margin 70%, an toàn nhiệt khi LED bật 15s.
#04 Không có cầu chì bảo vệ ngắn mạch Warning
Adapter 2A không tự cắt khi chập 5V↔GND. Trên breadboard hoặc dây dợ trong vỏ kín, một lần chập sẽ làm cháy dây/PCB hoặc phồng adapter.
V6: Thêm Polyfuse PPTC 1A tự phục hồi nối tiếp đường 5V vào ngay sau adapter.
#05 Tụ 470µF/10V — biên áp quá sát Warning
Đường 5V có xung peak 5.3-5.5V khi WiFi/camera bật. Tụ 10V chỉ có biên 2x. Sau thời gian, điện dung và áp chịu đựng giảm 20-30% → tụ phồng, rò chất điện phân.
V6: Đổi sang 470µF/16V hoặc 25V. Chênh giá không đáng kể, biên an toàn 3-5x.

esp32cam_security_v6.ino 
#include "esp_camera.h"
#include "SD_MMC.h"
#include "FS.h"
#include "time.h"
#include "WiFi.h"

// ===== PIN MAPPING V6 =====
#define PIR_PIN      13
#define IR_TX_PIN    12   // + pull-down 10k ngoài
#define IR_RX_PIN    33   // INPUT ONLY · Anode diode về đây

#define WIFI_SSID   "TEN_WIFI"
#define WIFI_PASS   "MAT_KHAU"
#define NTP_SERVER  "pool.ntp.org"
#define TZ_OFFSET   25200

// Camera AI-Thinker pins
#define PWDN_GPIO_NUM   32
#define RESET_GPIO_NUM  -1
#define XCLK_GPIO_NUM    0
#define SIOD_GPIO_NUM   26
#define SIOC_GPIO_NUM   27
#define Y9_GPIO_NUM     35
#define Y8_GPIO_NUM     34
#define Y7_GPIO_NUM     39
#define Y6_GPIO_NUM     36
#define Y5_GPIO_NUM     21
#define Y4_GPIO_NUM     19
#define Y3_GPIO_NUM     18
#define Y2_GPIO_NUM      5
#define VSYNC_GPIO_NUM  25
#define HREF_GPIO_NUM   23
#define PCLK_GPIO_NUM   22

#define RECORD_SECONDS   15
#define COOLDOWN_MS    5000
#define NIGHT_HOUR_START 18
#define NIGHT_HOUR_END    6
#define MAX_FILE_INDEX 9999

unsigned long lastTriggerMs = 0;
int  fileIndex  = 0;
bool timesynced = false;

bool initCamera() {
  camera_config_t cfg;
  cfg.ledc_channel = LEDC_CHANNEL_0; cfg.ledc_timer = LEDC_TIMER_0;
  cfg.pin_d0=Y2_GPIO_NUM; cfg.pin_d1=Y3_GPIO_NUM; cfg.pin_d2=Y4_GPIO_NUM;
  cfg.pin_d3=Y5_GPIO_NUM; cfg.pin_d4=Y6_GPIO_NUM; cfg.pin_d5=Y7_GPIO_NUM;
  cfg.pin_d6=Y8_GPIO_NUM; cfg.pin_d7=Y9_GPIO_NUM;
  cfg.pin_xclk=XCLK_GPIO_NUM; cfg.pin_pclk=PCLK_GPIO_NUM;
  cfg.pin_vsync=VSYNC_GPIO_NUM; cfg.pin_href=HREF_GPIO_NUM;
  cfg.pin_sscb_sda=SIOD_GPIO_NUM; cfg.pin_sscb_scl=SIOC_GPIO_NUM;
  cfg.pin_pwdn=PWDN_GPIO_NUM; cfg.pin_reset=RESET_GPIO_NUM;
  cfg.xclk_freq_hz = 20000000;
  cfg.pixel_format = PIXFORMAT_JPEG;
  cfg.jpeg_quality = 12;
  cfg.grab_mode    = CAMERA_GRAB_LATEST;

  if (psramFound()) {
    cfg.fb_count = 2; cfg.frame_size = FRAMESIZE_VGA;
  } else {
    cfg.fb_count = 1; cfg.frame_size = FRAMESIZE_SVGA;
  }
  if (esp_camera_init(&cfg) != ESP_OK) return false;
  sensor_t *s = esp_camera_sensor_get();
  s->set_brightness(s, 1);
  s->set_saturation(s, -1);
  return true;
}

void setNightVision(bool on) {
  digitalWrite(IR_TX_PIN, on ? HIGH : LOW);
  sensor_t *s = esp_camera_sensor_get();
  s->set_special_effect(s, on ? 2 : 0);
  s->set_agc_gain(s, on ? 8 : 0);
}

void recordVideo(int seconds) {
  if (fileIndex > MAX_FILE_INDEX) return;
  uint64_t freeBytes = SD_MMC.totalBytes() - SD_MMC.usedBytes();
  if (freeBytes < 5000000ULL) return;

  char path[32];
  sprintf(path, "/vid_%04d.mjpeg", fileIndex++);
  File f = SD_MMC.open(path, FILE_WRITE);
  if (!f) return;

  unsigned long endMs = millis() + seconds * 1000UL;
  int nullCount = 0;
  while (millis() < endMs) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) {
      if (++nullCount > 50) break;
      delay(10); continue;
    }
    nullCount = 0;
    uint32_t len = fb->len;
    f.write((uint8_t*)&len, 4);
    f.write(fb->buf, fb->len);
    esp_camera_fb_return(fb);
  }
  f.close();
}

bool isNightTime() {
  if (!timesynced) return true;
  struct tm ti;
  if (!getLocalTime(&ti)) return true;
  int h = ti.tm_hour;
  return (h >= NIGHT_HOUR_START || h < NIGHT_HOUR_END);
}

void setup() {
  Serial.begin(115200);
  delay(800);

  pinMode(PIR_PIN, INPUT);
  pinMode(IR_RX_PIN, INPUT);
  pinMode(IR_TX_PIN, OUTPUT);
  setNightVision(false);

  WiFi.begin(WIFI_SSID, WIFI_PASS);
  unsigned long wt = millis();
  while (WiFi.status() != WL_CONNECTED && millis()-wt < 8000) delay(300);

  if (WiFi.status() == WL_CONNECTED) {
    configTime(TZ_OFFSET, 0, NTP_SERVER);
    delay(2000);
    struct tm ti;
    timesynced = getLocalTime(&ti);
    WiFi.disconnect(true);
  }

  if (!initCamera()) {
    while(1) delay(1000);
  }
  SD_MMC.begin("/sdcard", true);
  Serial.println("V6 READY");
}

void loop() {
  bool motion  = digitalRead(PIR_PIN);
  bool irBreak = !digitalRead(IR_RX_PIN);

  if ((motion || irBreak) && (millis()-lastTriggerMs > COOLDOWN_MS)) {
    lastTriggerMs = millis();
    if (isNightTime()) setNightVision(true);
    delay(200);
    recordVideo(RECORD_SECONDS);
    setNightVision(false);
  }
  delay(100);
}
