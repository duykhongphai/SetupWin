🆕 VẤN ĐỀ MỚI 1: GPIO 33 trên AI-Thinker là INPUT ONLY — không thể dùng OUTPUT MỚI · CẦN XÁC NHẬN
V4 dùng GPIO 33 làm IR_RX_PIN (INPUT) — điều này hoàn toàn đúng vì GPIO 33 là input-only pin. Nhưng cần xác nhận rõ trong bảng đấu nối: GPIO 33 CHỈ đọc vào, KHÔNG xuất ra được. Code V4 đã đúng (INPUT), sơ đồ cũng đúng nhưng ghi chú chưa rõ cho người mới.
✅ V5: Ghi chú rõ "INPUT ONLY — chỉ nhận tín hiệu, không xuất" trong sơ đồ và bảng.
🔥 LỖI MỚI 2: Sơ đồ V4 — Chiều Diode Wired-OR vẽ NGƯỢC logic CRITICAL · SAI LOGIC
Trong SVG V4, mạch RX vẽ: Anode (mũi tên) hướng về phía module, Cathode (vạch) hướng về GPIO 33. Với module VS1838B Active-LOW: khi phát hiện, OUT xuống LOW (gần GND). Để Diode dẫn kéo GPIO 33 xuống LOW, chiều ĐÚNG phải là: Anode về GPIO 33, Cathode về module. V4 vẽ ngược → logic không hoạt động, GPIO 33 không bao giờ bị kéo xuống.
✅ V5: Vẽ lại đúng chiều: Cathode (vạch đen) → về phía GPIO 33 · Anode (đầu mũi tên) → về phía module OUT. Khi module kéo LOW, dòng chạy từ pull-up 3.3V → qua Diode (xuôi) → xuống module → GND.
💥 LỖI MỚI 3: Không có GND chung rõ ràng giữa tất cả khối mạch CRITICAL · MẠCH HỞ
Sơ đồ V4 vẽ từng khối GND riêng lẻ (PIR GND, TX GND, RX GND, Power GND). Không có đường nối hoặc ký hiệu rõ ràng rằng tất cả GND này phải được nối chung thành 1 điểm duy nhất. Người mới sẽ không nối GND các module với nhau → mạch hở, tín hiệu lơ lửng, nhiễu nặng.
✅ V5: Thêm thanh GND bus chung nổi bật, kẻ đường rõ ràng từ mỗi khối về thanh GND chung.
⚠️ LỖI MỚI 4: Code — configTime() bị thiếu → isNightTime() luôn return true WARNING · BUG LOGIC
Hàm isNightTime() gọi getLocalTime(&ti) nhưng setup() không hề gọi configTime() để đồng bộ NTP. Kết quả: getLocalTime() luôn thất bại → hàm luôn return true (luôn là đêm) → đèn IR luôn bật ban ngày, tốn điện, lộ camera.
✅ V5: Thêm configTime() vào setup(). Nếu không có WiFi, flag nightMode mặc định theo giờ khởi động.
⚠️ LỖI MỚI 5: Code — camera_fb_t* fb có thể null liên tục gây watchdog reset WARNING · CRASH
Trong recordVideo(), vòng while liên tục gọi esp_camera_fb_get(). Nếu camera bị lỗi hoặc bộ nhớ frame buffer cạn, fb trả về NULL → vòng lặp xoay liên tục không delay → WDT (Watchdog Timer) reset board sau ~3 giây.
✅ V5: Thêm delay(10) trong vòng lặp và đếm số lần NULL liên tiếp, nếu >50 lần thì thoát vòng lặp.

#include "esp_camera.h"
#include "SD_MMC.h"
#include "FS.h"
#include "time.h"
#include "WiFi.h"   // ← V5: cần cho configTime NTP

// ============================================================
//  PIN MAPPING — V5 FINAL
// ============================================================
#define PIR_PIN      13   // PIR AM312 OUT
#define IR_TX_PIN    12   // Base S8050 (+ pull-down 10k ngoài)
#define IR_RX_PIN    33   // GPIO 33 — INPUT ONLY, Wired-OR Diode
                              // Chiều Diode: Anode→module, Cathode→GPIO33

// WiFi (để đồng bộ giờ NTP — có thể bỏ nếu không dùng)
#define WIFI_SSID   "TEN_WIFI"
#define WIFI_PASS   "MAT_KHAU"
#define NTP_SERVER  "pool.ntp.org"
#define TZ_OFFSET   25200   // UTC+7 Việt Nam (7×3600)

// Camera pins – AI-Thinker
#define PWDN_GPIO_NUM   32
#define RESET_GPIO_NUM  -1
#define XCLK_GPIO_NUM    0
#define SIOD_GPIO_NUM   26
#define SIOC_GPIO_NUM   27
#define Y9_GPIO_NUM     35
#define Y8_GPIO_NUM     34
#define Y7_GPIO_NUM     39
#define Y6_GPIO_NUM     36
#define Y5_GPIO_NUM     21
#define Y4_GPIO_NUM     19
#define Y3_GPIO_NUM     18
#define Y2_GPIO_NUM      5
#define VSYNC_GPIO_NUM  25
#define HREF_GPIO_NUM   23
#define PCLK_GPIO_NUM   22

#define RECORD_SECONDS   15
#define COOLDOWN_MS    5000
#define NIGHT_HOUR_START 18
#define NIGHT_HOUR_END    6
#define MAX_FILE_INDEX 9999

unsigned long lastTriggerMs = 0;
int  fileIndex  = 0;
bool timesynced = false;  // ← V5: cờ biết có giờ chính xác không

bool initCamera() {
  camera_config_t cfg;
  cfg.ledc_channel = LEDC_CHANNEL_0; cfg.ledc_timer = LEDC_TIMER_0;
  cfg.pin_d0=Y2_GPIO_NUM; cfg.pin_d1=Y3_GPIO_NUM; cfg.pin_d2=Y4_GPIO_NUM;
  cfg.pin_d3=Y5_GPIO_NUM; cfg.pin_d4=Y6_GPIO_NUM; cfg.pin_d5=Y7_GPIO_NUM;
  cfg.pin_d6=Y8_GPIO_NUM; cfg.pin_d7=Y9_GPIO_NUM;
  cfg.pin_xclk=XCLK_GPIO_NUM; cfg.pin_pclk=PCLK_GPIO_NUM;
  cfg.pin_vsync=VSYNC_GPIO_NUM; cfg.pin_href=HREF_GPIO_NUM;
  cfg.pin_sscb_sda=SIOD_GPIO_NUM; cfg.pin_sscb_scl=SIOC_GPIO_NUM;
  cfg.pin_pwdn=PWDN_GPIO_NUM; cfg.pin_reset=RESET_GPIO_NUM;
  cfg.xclk_freq_hz = 20000000;
  cfg.pixel_format = PIXFORMAT_JPEG;
  cfg.jpeg_quality = 12;
  cfg.grab_mode    = CAMERA_GRAB_LATEST;
  if (psramFound()) {
    cfg.fb_count   = 2;
    cfg.frame_size = FRAMESIZE_VGA;
    Serial.println("PSRAM OK");
  } else {
    cfg.fb_count   = 1;
    cfg.frame_size = FRAMESIZE_SVGA;
    Serial.println("No PSRAM — fallback SVGA");
  }
  if (esp_camera_init(&cfg) != ESP_OK) return false;
  sensor_t *s = esp_camera_sensor_get();
  s->set_brightness(s, 1);
  s->set_saturation(s, -1);
  return true;
}

void setNightVision(bool enable) {
  digitalWrite(IR_TX_PIN, enable ? HIGH : LOW);
  sensor_t *s = esp_camera_sensor_get();
  s->set_special_effect(s, enable ? 2 : 0);
  s->set_agc_gain(s, enable ? 8 : 0);
}

void recordVideo(int seconds) {
  if (fileIndex > MAX_FILE_INDEX) { Serial.println("File limit!"); return; }
  uint64_t freeBytes = SD_MMC.totalBytes() - SD_MMC.usedBytes();
  if (freeBytes < 5000000ULL) { Serial.println("SD full!"); return; }

  char path[32];
  sprintf(path, "/vid_%04d.mjpeg", fileIndex++);
  File f = SD_MMC.open(path, FILE_WRITE);
  if (!f) { Serial.println("SD open fail!"); return; }

  unsigned long endMs = millis() + (unsigned long)seconds * 1000UL;
  int nullCount = 0;  // ← V5: đếm số lần fb == NULL liên tiếp
  while (millis() < endMs) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) {
      nullCount++;
      if (nullCount > 50) {  // Camera lỗi liên tục → thoát tránh WDT reset
        Serial.println("Camera fb null x50 — abort recording");
        break;
      }
      delay(10);  // ← Nhường CPU, tránh WDT
      continue;
    }
    nullCount = 0;  // Reset counter khi có frame
    uint32_t len = fb->len;
    f.write((uint8_t*)&len, 4);
    f.write(fb->buf, fb->len);
    esp_camera_fb_return(fb);
  }
  f.close();
}

// ← V5: Có đồng hồ thực → dựa vào giờ. Không có → mặc định ban đêm
bool isNightTime() {
  if (!timesynced) return true;  // Không có WiFi/NTP → luôn bật IR
  struct tm ti;
  if (!getLocalTime(&ti)) return true;
  int h = ti.tm_hour;
  return (h >= NIGHT_HOUR_START || h < NIGHT_HOUR_END);
}

void setup() {
  Serial.begin(115200);
  delay(800);

  pinMode(PIR_PIN,   INPUT);
  pinMode(IR_RX_PIN, INPUT);  // GPIO33 INPUT ONLY — pull-up 10k lắp ngoài
  pinMode(IR_TX_PIN, OUTPUT);
  setNightVision(false);
  // Pull-down 10kΩ cho GPIO12 đã lắp ngoài phần cứng

  // ← V5: Kết nối WiFi để đồng bộ giờ
  Serial.printf("Kết nối WiFi %s...\n", WIFI_SSID);
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  unsigned long wt = millis();
  while (WiFi.status() != WL_CONNECTED && millis()-wt < 8000) {
    delay(500); Serial.print(".");
  }
  if (WiFi.status() == WL_CONNECTED) {
    configTime(TZ_OFFSET, 0, NTP_SERVER);  // UTC+7
    delay(2000);
    struct tm ti;
    timesynced = getLocalTime(&ti);
    Serial.printf("\nGiờ: %02d:%02d %s\n", ti.tm_hour, ti.tm_min,
                  timesynced ? "(đồng bộ OK)" : "(lỗi NTP)");
    WiFi.disconnect(true);  // Ngắt WiFi tiết kiệm điện
  } else {
    Serial.println("\nKhông có WiFi — IR luôn bật");
  }

  if (!initCamera()) {
    Serial.println("FATAL: Camera fail!");
    while(1) delay(1000);
  }
  if (!SD_MMC.begin("/sdcard", true)) {
    Serial.println("WARNING: SD fail!");
  } else {
    Serial.printf("SD OK: %llu MB free\n",
      (SD_MMC.totalBytes()-SD_MMC.usedBytes())/1048576ULL);
  }
  Serial.println("=== V5 READY ===");
}

void loop() {
  bool motion  = digitalRead(PIR_PIN);
  bool irBreak = !digitalRead(IR_RX_PIN);  // Diode kéo LOW = có xâm nhập

  if ((motion || irBreak) && (millis()-lastTriggerMs > COOLDOWN_MS)) {
    lastTriggerMs = millis();
    Serial.printf("TRIGGER PIR=%d IR=%d\n", motion, irBreak);

    if (isNightTime()) setNightVision(true);
    delay(200);
    recordVideo(RECORD_SECONDS);
    setNightVision(false);
  }
  delay(100);
}
