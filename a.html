✅ Thiết kế Tối Ưu: Gộp 3 LED TX vào 1 Transistor (dư sức tải). Gộp 3 RX qua Diode Logic (tiết kiệm chân, chống nhiễu). Tránh xa hoàn toàn các chân cấm kỵ (0, 2, 4, 14, 15, 16). Bổ sung Tụ Hóa bắt buộc.
⚠️ Phân tích sự cố: Không dùng GPIO 16 (chết bộ nhớ PSRAM camera), không dùng GPIO 4 (sáng đèn flash trắng gây lộ camera), không dùng GPIO 0 (chạm cảm biến lúc cắm điện sẽ gây treo máy). Giải pháp V3 đã né toàn bộ các chân này!

2. Khối LED Phát (IR TX) - Dùng chung 1 Transistor

💡 Mẹo Kỹ Sư: 3 bóng LED IR ăn khoảng 100mA. 1 con NPN S8050 chịu được tới 700mA. Không cần dùng 3 transistor phức tạp, chỉ cần nối song song 3 bóng LED (mỗi bóng nối 1 trở 100Ω riêng) vào chân C của 1 transistor là chạy hoàn hảo, lại tiết kiệm được 2 chân GPIO.

3. Khối LED Thu (IR RX) - Mạch Diode "Wired-OR"

💡 Diode Logic Gate: Cả 3 module nhận hồng ngoại được nối chụm vào 1 chân ESP32 thông qua 3 Diode. Đầu vạch đen (Cathode) hướng về phía mắt thu. Nếu BẤT CỨ mắt thu nào phát hiện xâm nhập (hạ xuống LOW), Diode sẽ mở và kéo GPIO 3 xuống mức LOW. Không bị chập mạch, cực kỳ an toàn!

// Code tinh gọn, chuẩn logic phần cứng V3, bỏ qua các chân rủi ro.
#include "esp_camera.h"
#include "SD_MMC.h"
#include "FS.h"
#include "time.h"

// ============================================================
//  PIN MAPPING - BẢN V3 (AN TOÀN TUYỆT ĐỐI)
// ============================================================
#define PIR_PIN     13   // Cảm biến thân nhiệt
#define IR_TX_PIN   12   // Bật toàn bộ LED phát sáng ban đêm
#define IR_RX_PIN    3   // Chân nhận tín hiệu chụm từ 3 cảm biến RX

// Camera pins – AI-Thinker
#define PWDN_GPIO_NUM  32
#define RESET_GPIO_NUM -1
#define XCLK_GPIO_NUM   0
#define SIOD_GPIO_NUM  26
#define SIOC_GPIO_NUM  27
#define Y9_GPIO_NUM    35
#define Y8_GPIO_NUM    34
#define Y7_GPIO_NUM    39
#define Y6_GPIO_NUM    36
#define Y5_GPIO_NUM    21
#define Y4_GPIO_NUM    19
#define Y3_GPIO_NUM    18
#define Y2_GPIO_NUM     5
#define VSYNC_GPIO_NUM 25
#define HREF_GPIO_NUM  23
#define PCLK_GPIO_NUM  22

// ============================================================
//  CONFIG
// ============================================================
#define RECORD_SECONDS    15   // giây quay mỗi lần
#define COOLDOWN_MS     5000   // ms nghỉ để thẻ nhớ xả đệm
#define NIGHT_HOUR_START  18   // giờ bật hồng ngoại (18:00)
#define NIGHT_HOUR_END     6   // giờ tắt hồng ngoại (6:00)

unsigned long lastTriggerMs = 0;
int fileIndex = 0;

// ============================================================
//  FUNCTIONS
// ============================================================
bool initCamera() {
  camera_config_t cfg;
  cfg.ledc_channel = LEDC_CHANNEL_0;
  cfg.ledc_timer   = LEDC_TIMER_0;
  cfg.pin_d0       = Y2_GPIO_NUM;
  cfg.pin_d1       = Y3_GPIO_NUM;
  cfg.pin_d2       = Y4_GPIO_NUM;
  cfg.pin_d3       = Y5_GPIO_NUM;
  cfg.pin_d4       = Y6_GPIO_NUM;
  cfg.pin_d5       = Y7_GPIO_NUM;
  cfg.pin_d6       = Y8_GPIO_NUM;
  cfg.pin_d7       = Y9_GPIO_NUM;
  cfg.pin_xclk     = XCLK_GPIO_NUM;
  cfg.pin_pclk     = PCLK_GPIO_NUM;
  cfg.pin_vsync    = VSYNC_GPIO_NUM;
  cfg.pin_href     = HREF_GPIO_NUM;
  cfg.pin_sscb_sda = SIOD_GPIO_NUM;
  cfg.pin_sscb_scl = SIOC_GPIO_NUM;
  cfg.pin_pwdn     = PWDN_GPIO_NUM;
  cfg.pin_reset    = RESET_GPIO_NUM;
  cfg.xclk_freq_hz = 20000000;
  cfg.pixel_format = PIXFORMAT_JPEG;
  cfg.frame_size   = FRAMESIZE_VGA;  
  cfg.jpeg_quality = 12;             
  cfg.fb_count     = 2;              // Cần có PSRAM để dùng 2 frame
  cfg.grab_mode    = CAMERA_GRAB_LATEST;

  if (esp_camera_init(&cfg) != ESP_OK) return false;
  
  sensor_t *s = esp_camera_sensor_get();
  s->set_brightness(s, 1);
  s->set_saturation(s, -1);
  return true;
}

void setNightVision(bool enable) {
  digitalWrite(IR_TX_PIN, enable ? HIGH : LOW);
  sensor_t *s = esp_camera_sensor_get();
  if (enable) {
    s->set_special_effect(s, 2); // Chế độ đen trắng chống nhiễu
    s->set_agc_gain(s, 8);       // Tăng sáng nhạy cảm
  } else {
    s->set_special_effect(s, 0); // Trở lại có màu
    s->set_agc_gain(s, 0);
  }
}

void recordVideo(int seconds) {
  char path[32];
  sprintf(path, "/vid_%04d.mjpeg", fileIndex++);
  File f = SD_MMC.open(path, FILE_WRITE);
  if (!f) return;
  
  unsigned long endMs = millis() + (unsigned long)seconds * 1000;
  while (millis() < endMs) {
    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) continue;
    uint32_t len = fb->len;
    f.write((uint8_t*)&len, 4);
    f.write(fb->buf, fb->len);
    esp_camera_fb_return(fb);
  }
  f.close();
}

bool isNightTime() {
  struct tm ti;
  if (!getLocalTime(&ti)) return true; // Mặc định là đêm nếu mất giờ
  int h = ti.tm_hour;
  return (h >= NIGHT_HOUR_START || h < NIGHT_HOUR_END);
}

// ============================================================
//  MAIN LOOP
// ============================================================
void setup() {
  Serial.begin(115200);
  delay(1000);

  // Cấu hình chân
  pinMode(PIR_PIN, INPUT);
  pinMode(IR_RX_PIN, INPUT); // Pull-up 10k đã lắp bằng phần cứng ngoài
  pinMode(IR_TX_PIN, OUTPUT);
  setNightVision(false);

  if (!initCamera()) {
    Serial.println("Camera Lỗi! Ktra PSRAM / Dây cắm");
    while(1);
  }

  if (!SD_MMC.begin("/sdcard", true)) { 
    Serial.println("Lỗi Thẻ Nhớ!");
  } else {
    Serial.println("Thẻ Nhớ OK");
  }
}

void loop() {
  bool motion_detected = digitalRead(PIR_PIN);       
  bool beam_broken     = !digitalRead(IR_RX_PIN); // LOW tức là có ngắt tia do Diode Logic

  if ((motion_detected || beam_broken) && (millis() - lastTriggerMs > COOLDOWN_MS)) {
    lastTriggerMs = millis();
    
    if (isNightTime()) setNightVision(true);
    delay(200); // Đợi cảm biến ảnh cân bằng sáng

    recordVideo(RECORD_SECONDS);
    
    setNightVision(false); // Quay xong tắt đèn tiết kiệm điện
  }
  delay(100);
}
